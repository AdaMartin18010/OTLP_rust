## 🚀 部署和使用

### 快速开始

```rust
use otlp::{OtlpClient, OtlpConfig};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let config = OtlpConfig::default();
    let client = OtlpClient::new(config).await?;
    
    // 发送遥测数据
    client.send_batch(vec![telemetry_data]).await?;
    
    Ok(())
}
```
