impl OpaClient {
    pub async fn evaluate_policy(
        &self,
        policy_path: &str,
        input: serde_json::Value,
    ) -> Result<PolicyDecision> {

            allow: decision.result.as_bool().unwrap_or(false),
            result: decision.result,
